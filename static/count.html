计数器示例

const client = redis.createClient(6379, 'localhost');

//初始化计数器
//set goodCount 0
client.set("goodCount", '0');

const getA = promisify(client.get).bind(client);

//相当于 get goodCount
async function getCount() {
    return getA("goodCount");
}
//相当于 incr goodCount
function addCount() {
    client.incr("goodCount");
}
